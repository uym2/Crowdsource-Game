USE uym2$theimgame2016;
DROP procedure IF EXISTS `sp_postObjMark`;

DELIMITER $$
CREATE PROCEDURE `sp_postObjMark`(
	IN p_userID bigint,
	IN p_imgID bigint,
	IN p_topleftX int,
	IN p_topleftY int,
	IN p_bottomrightX int,
	IN p_bottomrightY int,
	IN p_trustFactor float
)
BEGIN
	INSERT INTO tbl_objmark(
	user_id,
	img_id,
	topleft_x,
	topleft_y,
	bottomright_x,
	bottomright_y,
	trust_factor
	)
	values(
	p_userID,
	p_imgID,
	p_topleftX,
	p_topleftY,
	p_bottomrightX,
	p_bottomrightY,
	p_trustFactor
	);
END$$

DELIMITER ;
